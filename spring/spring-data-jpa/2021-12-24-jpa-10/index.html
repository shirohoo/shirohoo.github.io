@Table
@Entity
public class Member {

    @Id
    @GeneratedValue
    private Long id;

    private String name;

    @OneToOne(
        mappedBy = "member",
        fetch = FetchType.LAZY,
        cascade = CascadeType.ALL,
        optional = false
    )
    private Locker locker;

    public void setLocker(Locker locker) {
        if (locker == null) {
            if (this.locker != null) {
                this.locker.setMember(null);
            }
        }
        else {
            locker.setMember(this);
        }
        this.locker = locker;
    }

}
